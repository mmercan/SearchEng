Angular Components

Service not a directive

Add "notificationService" service in to your controller dependencies

angular.module("app").controller("angularcomp", ["$scope", "$http", "$routeParams", "$location" ,"alertService", "notificationService","mainMenuService", function($scope, $http, $routeParams, $location, alertservice, notificationservice,mainMenuService)
NameTypeActionDesciption
notificationService.success function (title,message,options)
notificationService.info function (title,message,options)
notificationService.warning function (title,message,options)
notificationService.error function (title,message,options)
title string/html title on notifation
message string/html content on notifation
options "closeButton": false,
"debug": false,
"positionClass": "toast-top-right",
"onclick": null,
"showDuration": "3000",
"hideDuration": "1000",
"timeOut": "5000",
"extendedTimeOut": "1000",
"showEasing": "swing",
"hideEasing": "linear",
"showMethod": "fadeIn",
"hideMethod": "fadeOut"
